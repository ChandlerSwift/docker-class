Sample Software


void setup() {
    // Setup goes here!
    pinMode(2, OUTPUT);
    pinMode(3, INPUT_PULLUP);

    // Initializing:
    for (int i = 0; i < 5; i++) {
        digitalWrite(2, HIGH);
        delay(500);
        digitalWrite(2, LOW);
        delay(500);
    }
}
void loop() {
    digitalWrite(2, !digitalRead(3));
}
(see also: blink.ino)

Equivalently…


#include <Arduino.h>
int main() {
    // Setup goes here!
    pinMode(2, OUTPUT);
    pinMode(3, INPUT_PULLUP);

    // Initializing:
    for (int i = 0; i < 5; i++) {
        digitalWrite(2, HIGH);
        delay(500);
        digitalWrite(2, LOW);
        delay(500);
    }

    while (true) (
        digitalWrite(2, !digitalRead(3));
    }
}

Pseudocode

  • Wait random time (checking for false presses)
